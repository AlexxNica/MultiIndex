235  coarse_vocabs_matrices_.resize(coarse_vocabs_.size());
236  coarse_centroids_norms_.resize(coarse_vocabs_.size(), vector<float>(coarse_vocabs_[0].size()));
237  for(int coarse_id = 0; coarse_id < coarse_vocabs_matrices_.size(); ++coarse_id) {
238  coarse_vocabs_matrices_[coarse_id] = new float[coarse_vocabs_[0].size() * coarse_vocabs_[0][0].size()];
239  for(int i = 0; i < coarse_vocabs_[0].size(); ++i) {
240  Coord norm = 0;

242  coarse_vocabs_matrices_[coarse_id][coarse_vocabs_[0][0].size() * i + j] = coarse_vocabs_[coarse_id][i][j];
243  norm += coarse_vocabs_[coarse_id][i][j] * coarse_vocabs_[coarse_id][i][j];
244  }
245  coarse_centroids_norms_[coarse_id][i] = norm;
246  }
247  }
248  products_ = new Coord[coarse_vocabs_[0].size()];
249  query_norms_.resize(coarse_vocabs_[0].size());
250  residual_ = new Coord[coarse_vocabs_[0][0].size() * coarse_vocabs_.size()];

258 template<class Record, class MetaInfo>
259 void MultiSearcher<Record, MetaInfo>::GetNearestSubspacesCentroids(const Point& point,
260  const int subspace_centroins_count,
261  vector<NearestSubspaceCentroids>*
262  subspaces_short_lists) const {
263  subspaces_short_lists->resize(coarse_vocabs_.size());
264  Dimensions subspace_dimension = point.size() / coarse_vocabs_.size();
265  for(int subspace_index = 0; subspace_index < coarse_vocabs_.size(); ++subspace_index) {
266  Dimensions start_dim = subspace_index * subspace_dimension;
267  Dimensions final_dim = std::min((Dimensions)point.size(), start_dim + subspace_dimension);
268  Coord query_norm = cblas_sdot(final_dim - start_dim, &(point[start_dim]), 1, &(point[start_dim]), 1);
269  std::fill(query_norms_.begin(), query_norms_.end(), query_norm);
270  cblas_saxpy(coarse_vocabs_[0].size(), 1, &(coarse_centroids_norms_[subspace_index][0]), 1, &(query_norms_[0]), 1);
271  cblas_sgemv(CblasRowMajor, CblasNoTrans, coarse_vocabs_[0].size(), subspace_dimension, -2.0,
272  coarse_vocabs_matrices_[subspace_index], subspace_dimension, &(point[start_dim]), 1, 1, &(query_norms_[0]), 1);
273  subspaces_short_lists->at(subspace_index).resize(query_norms_.size());
274  for(int i = 0; i < query_norms_.size(); ++i) {
275  subspaces_short_lists->at(subspace_index)[i] = std::make_pair(query_norms_[i], i);
276  }
277  std::nth_element(subspaces_short_lists->at(subspace_index).begin(),
278  subspaces_short_lists->at(subspace_index).begin() + subspace_centroins_count,
279  subspaces_short_lists->at(subspace_index).end());
280  subspaces_short_lists->at(subspace_index).resize(subspace_centroins_count);
281  std::sort(subspaces_short_lists->at(subspace_index).begin(),
282  subspaces_short_lists->at(subspace_index).end());
283  }

298 template<class Record, class MetaInfo>
299 bool MultiSearcher<Record, MetaInfo>::TraverseNextMultiIndexCell(const Point& point,
300  vector<pair<Distance, MetaInfo> >*
301  nearest_subpoints) const {
302  MergedItemIndices cell_inner_indices;
303  clock_t before = clock();
304  if(!merger_.GetNextMergedItemIndices(&cell_inner_indices)) {
305  return false;
306  }
307  clock_t after = clock();
308  perf_tester_.cell_coordinates_time += after - before;
309  vector<int> cell_coordinates(cell_inner_indices.size());
310  for(int list_index = 0; list_index < merger_.lists_ptr->size(); ++list_index) {
311  cell_coordinates[list_index] = merger_.lists_ptr->at(list_index)[cell_inner_indices[list_index]].second;
312  }
313  int cell_start, cell_finish;
314  before = clock();
315  GetCellEdgesInMultiIndexArray(cell_coordinates, &cell_start, &cell_finish);
316  after = clock();
317  perf_tester_.cell_edges_time += after - before;
318  if(cell_start >= cell_finish) {
319  return true;
320  }
321  vector<Record>::const_iterator it = multiindex_.multiindex.begin() + cell_start;
322  GetResidual(point, cell_coordinates, coarse_vocabs_, residual_);
323  cell_finish = std::min((int)cell_finish, cell_start + (int)nearest_subpoints->size() - found_neghbours_count_);
324  for(int array_index = cell_start; array_index < cell_finish; ++array_index) {
325  if(rerank_mode_ == USE_RESIDUALS) {
326  RecordToMetainfoAndDistance<Record, MetaInfo>(residual_, *it,
327  &(nearest_subpoints->at(found_neghbours_count_)),
328  cell_coordinates, fine_vocabs_);
329  } else if(rerank_mode_ == USE_INIT_POINTS) {
330  RecordToMetainfoAndDistance<Record, MetaInfo>(&(point[0]), *it,
331  &(nearest_subpoints->at(found_neghbours_count_)),
332  cell_coordinates, fine_vocabs_);
333  }
334  perf_tester_.NextNeighbour();
335  ++found_neghbours_count_;
336  ++it;
337  }
338  return true;
339 }

342 template<class Record, class MetaInfo>
343 void MultiSearcher<Record, MetaInfo>::GetNearestNeighbours(const Point& point, int k,
344  vector<pair<Distance, MetaInfo> >* neighbours) const {
345  perf_tester_.handled_queries_count += 1;
346  neighbours->resize(k);
347  perf_tester_.ResetQuerywiseStatistic();
348  clock_t start = clock();
349  perf_tester_.search_start = start;
350  clock_t before = clock();
351  vector<NearestSubspaceCentroids> subspaces_short_lists;
352  GetNearestSubspacesCentroids(point, subspace_centroids_to_consider_, &subspaces_short_lists);
353  clock_t after = clock();
354  perf_tester_.nearest_subcentroids_time += after - before;
355  clock_t before_merger = clock();
356  merger_.setLists(subspaces_short_lists);
357  clock_t after_merger = clock();
358  perf_tester_.merger_init_time += after_merger - before_merger;
359  clock_t before_traversal = clock();
360  found_neghbours_count_ = 0;
361  bool traverse_next_cell = true;
362  int cells_visited = 0;
363  while(found_neghbours_count_ < k && traverse_next_cell) {
364  perf_tester_.cells_traversed += 1;
365  traverse_next_cell = TraverseNextMultiIndexCell(point, neighbours);
366  cells_visited += 1;
367  }
368  clock_t after_traversal = clock();
369  perf_tester_.full_traversal_time += after_traversal - before_traversal;
370  if(do_rerank_) {
371  if(neighbours->size() > 10000) {
372  std::nth_element(neighbours->begin(), neighbours->begin() + 10000, neighbours->end());
373  neighbours->resize(10000);
374  }
375  std::sort(neighbours->begin(), neighbours->end());
376  }
377  clock_t finish = clock();
378  perf_tester_.full_search_time += finish - start;
379 }

381 template<>
382 inline void RecordToMetainfoAndDistance<RerankADC8, PointId>(const Coord* point, const RerankADC8& record,
383  pair<Distance, PointId>* result,
384  const vector<int>& cell_coordinates,
385  const vector<Centroids>& fine_vocabs) {
386  result->second = record.pid;
387  int coarse_clusters_count = cell_coordinates.size();
388  int fine_clusters_count = fine_vocabs.size();
389  int coarse_to_fine_ratio = fine_clusters_count / coarse_clusters_count;
390  int subvectors_dim = SPACE_DIMENSION / fine_clusters_count;
391  char* rerank_info_ptr = (char*)&record + sizeof(record.pid);
392  for(int centroid_index = 0; centroid_index < fine_clusters_count; ++centroid_index) {
393  int start_dim = centroid_index * subvectors_dim;
394  int final_dim = start_dim + subvectors_dim;
395  FineClusterId pid_nearest_centroid = *((FineClusterId*)rerank_info_ptr);
396  rerank_info_ptr += sizeof(FineClusterId);
397  int current_coarse_index = centroid_index / coarse_to_fine_ratio;
398  Distance subvector_distance = 0;
399  for(int i = start_dim; i < final_dim; ++i) {
400  Coord diff = fine_vocabs[centroid_index][pid_nearest_centroid][i - start_dim] - point[i];
401  subvector_distance += diff * diff;
402  }
403  result->first += subvector_distance;
404  }
405 }
